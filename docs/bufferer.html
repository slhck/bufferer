212    def _generate_loop_cmds(self):
213        """
214        Construct the looping commands
215        """
216
217        vloop_cmds = []
218        aloop_cmds = []
219        venable_cmds = []
220        aenable_cmds = []
221
222        total_vlooped = 0
223        total_alooped = 0
224        total_buf_len = 0
225
226        self.enable_black_cmd = None
227
228        # trim_cmds are only used for freeze
229        trim_cmds = []
230        last_buf_end = 0
231
232        for buf_event in self.buflist:
233            buf_pos, buf_len = buf_event
234            buf_pos_enable = round(total_buf_len + buf_pos, 3)
235            buf_len_enable = round(buf_pos_enable + buf_len, 3)
236
237            # FIXME: the enable time is slightly smaller than what one would expect, with video
238            buf_len_enable_video = buf_len_enable - 0.001
239
240            total_buf_len = total_buf_len + buf_len
241
242            if self.has_video:
243                # offset buf_position by the total number of looped frames
244                buf_pos_frames = int(self.fps * buf_pos) + total_vlooped
245                # FIXME: the number of frames needs to be 1 shorter?
246                buf_len_frames = int(self.fps * buf_len)
247
248                loop_cmd = f"loop=loop={buf_len_frames}:size=1:start={buf_pos_frames},setpts=N/FRAME_RATE/TB"
249                vloop_cmds.append(loop_cmd)
250
251                total_vlooped += buf_len_frames
252
253                venable_cmd = f"between(t,{buf_pos_enable},{buf_len_enable_video})"
254                venable_cmds.append(venable_cmd)
255
256                trim_cmd = f"trim=start_frame={last_buf_end}:end_frame={buf_pos_frames + buf_len_frames},setpts=PTS-STARTPTS"
257                trim_cmds.append(trim_cmd)
258                last_buf_end = buf_pos_frames + 2 * buf_len_frames
259
260            if self.has_audio:
261                # offset buf_position by the total number of looped samples
262                buf_pos_samples = int(self.samplerate * buf_pos) + total_alooped
263                buf_len_samples = int(self.samplerate * buf_len)
264
265                aloop_cmd = f"aloop=loop={buf_len_samples}:size=1:start={buf_pos_samples},asetpts=N/SAMPLE_RATE/TB"
266                aloop_cmds.append(aloop_cmd)
267
268                total_alooped += buf_len_samples
269
270                aenable_cmd = f"between(t,{buf_pos_enable},{buf_len_enable})"
271                aenable_cmds.append(aenable_cmd)
272
273            if int(buf_pos_enable) == 0:
274                self.enable_black_cmd = f"between(t,0,{buf_len_enable_video})"
275
276        # needs an extra trim at the end to get the end of the file
277        if self.fps is None:
278            raise RuntimeError("fps not specified!")
279
280        duration_in_frames = (
281            int(self._get_duration_in_seconds() * self.fps) + total_vlooped
282        )
283        trim_cmd = f"trim=start_frame={last_buf_end}:end_frame={duration_in_frames},setpts=PTS-STARTPTS"
284        trim_cmds.append(trim_cmd)
285
286        self.vloop_cmd = (",").join(vloop_cmds)
287        self.aloop_cmd = (",").join(aloop_cmds)
288        self.trim_cmds = trim_cmds
289        self.venable_cmd = ("+").join(venable_cmds)
290        self.aenable_cmd = ("+").join(aenable_cmds)

363    def trim_video(self):
364        """
365        Remove frames after the frozen, repeated, ones to emulate freezing with skipping
366        """
367        trim_extra_frames = [
368            self.ffmpeg_path,
369            self.overwrite_spec,
370        ]
371
372        trim_extra_frames.extend(
373            [
374                "-i",
375                self._get_tmp_filename("video"),
376            ]
377        )
378
379        vfilters = []
380        filter_interface_list = []
381
382        for ii in range(0, len(self.trim_cmds)):
383            filter_interface_list.append(f"[i{ii}v]")
384
385        remaining_trim_cmds = list(self.trim_cmds)
386        for ii, jj in enumerate(filter_interface_list):
387            filter_string = f"[0:v]{remaining_trim_cmds.pop(0)}{jj}"
388            vfilters.append(filter_string)
389        filters = ";".join(vfilters) + ";"
390        filters += "".join(filter_interface_list)
391        filters += f"concat=n={len(self.trim_cmds)}:v=1[outv]"
392
393        trim_extra_frames.extend(["-filter_complex", filters])
394
395        trim_extra_frames.extend(["-map", "[outv]"])
396
397        trim_extra_frames.extend(
398            [
399                "-c:v",
400                self.vcodec,
401                "-vsync",
402                "cfr",
403                self._get_tmp_filename("skipping"),
404            ]
405        )
406
407        self.run_command(trim_extra_frames)

525    def _get_duration_in_seconds(self):
526        """
527        Convert between the HH:MM:SS.sss format, to total number of seconds.
528        """
529        if self.input_duration is None:
530            raise RuntimeError("Input duration not specified")
531        h, m, s = self.input_duration.split(":")
532        time_in_seconds = float(
533            datetime.timedelta(
534                hours=int(h), minutes=int(m), seconds=float(s)

547    def insert_buf_audiovisual(self):
548        """
549        Insert the buffering events on both audio and video tracks, looping the video
550        frames and audio samples at the corresponding positions.
551        """
552
553        self._generate_loop_cmds()
554        self._set_specs()
555
556        tmp_file_list = []
557
558        try:
559            if self.has_video:
560                logger.info("running command for processing video")
561                self.insert_buf_video()
562                tmp_file_list.append(self._get_tmp_filename("video"))
563            if self.skipping:
564                logger.info("running command for trimming video")
565                self.trim_video()
566                tmp_file_list.append(self._get_tmp_filename("skipping"))
567            else:
568                if self.has_audio:
569                    logger.info("running command for processing audio")
570                    self.insert_buf_audio()
571                    tmp_file_list.append(self._get_tmp_filename("audio"))
572            logger.info("running command for merging video/audio")
573            self.merge_audio_video()
574        except Exception as e:
575            logger.error(f"error running processing: {e}")
576        finally:
577            if not self.dry:
578                for file in tmp_file_list:
579                    if os.path.isfile(file):
580                        os.remove(file)
581                    else:
582                        logger.warn(f"temporary file {file} not found!")
