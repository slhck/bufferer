115        """
116        Parse various info from the input file
117        """
118
119        p = subprocess.Popen(
120            ["ffmpeg", "-i", self.input_file],
121            stdout=subprocess.PIPE,
122            stderr=subprocess.PIPE,
123        )
124        _, stderr = p.communicate()
125        output = stderr.decode("utf-8")
126
127        if video_match := re.compile(r"Video: (.*)").search(output):
128            self.has_video = True
129            video_line = video_match.group(1)
130            if fps_match := re.compile(r".*, ([0-9.]+) fps,.*").search(video_line):
131                self.fps = float(fps_match.group(1))
132
133            if video_resolution_match := re.compile(r".*, (\d+x\d+)[, ].*").search(
134                video_line
135            ):
136                self.video_resolution = video_resolution_match.group(1)
137
138        if (
139            audio_match := re.compile(r"Audio: (.*)").search(output)
140        ) and not self.audio_disable:
141            self.has_audio = True
142            audio_line = audio_match.group(1)
143            if hz_match := re.compile(r".*, ([0-9]+) Hz,.*").search(audio_line):
144                self.samplerate = float(hz_match.group(1))
145
146        if not (self.has_audio or self.has_video):
147            raise RuntimeError("file has no video or audio stream")
148
149        if input_duration_match := re.compile(
150            r".* Duration: ([0-9.]+:[0-9.]+:[0-9.]+\.[0-9.]+), .*"
151        ).search(output):
152            self.input_duration = input_duration_match.group(1)
153
154        if not self.fps:
155            raise RuntimeError("Could not detect video fps from input file!")
156        if self.has_audio and not self.samplerate:
157            raise RuntimeError("Could not detect audio sample rate from input file!")
158        if self.has_video and not self.video_resolution:
159            raise RuntimeError("Could not detect video resolution from input file!")
160        if not self.input_duration:
161            raise RuntimeError("Could not detect duration from input file!")

163    def _generate_loop_cmds(self):
164        """
165        Construct the looping commands
166        """
167
168        vloop_cmds = []
169        aloop_cmds = []
170        venable_cmds = []
171        aenable_cmds = []
172
173        total_vlooped = 0
174        total_alooped = 0
175        total_buf_len = 0
176
177        self.enable_black_cmd = None
178
179        # trim_cmds are only used for freeze
180        trim_cmds = []
181        last_buf_end = 0
182
183        for buf_event in self.buflist:
184            buf_pos, buf_len = buf_event
185            buf_pos_enable = round(total_buf_len + buf_pos, 3)
186            buf_len_enable = round(buf_pos_enable + buf_len, 3)
187
188            # FIXME: the enable time is slightly smaller than what one would expect, with video
189            buf_len_enable_video = buf_len_enable - 0.001
190
191            total_buf_len = total_buf_len + buf_len
192
193            if self.has_video:
194                # offset buf_position by the total number of looped frames
195                buf_pos_frames = int(self.fps * buf_pos) + total_vlooped
196                # FIXME: the number of frames needs to be 1 shorter?
197                buf_len_frames = int(self.fps * buf_len)
198
199                loop_cmd = f"loop=loop={buf_len_frames}:size=1:start={buf_pos_frames},setpts=N/FRAME_RATE/TB"
200                vloop_cmds.append(loop_cmd)
201
202                total_vlooped += buf_len_frames
203
204                venable_cmd = f"between(t,{buf_pos_enable},{buf_len_enable_video})"
205                venable_cmds.append(venable_cmd)
206
207                trim_cmd = f"trim=start_frame={last_buf_end}:end_frame={buf_pos_frames + buf_len_frames},setpts=PTS-STARTPTS"
208                trim_cmds.append(trim_cmd)
209                last_buf_end = buf_pos_frames + 2 * buf_len_frames
210
211            if self.has_audio:
212                # offset buf_position by the total number of looped samples
213                buf_pos_samples = int(self.samplerate * buf_pos) + total_alooped
214                buf_len_samples = int(self.samplerate * buf_len)
215
216                aloop_cmd = f"aloop=loop={buf_len_samples}:size=1:start={buf_pos_samples},asetpts=N/SAMPLE_RATE/TB"
217                aloop_cmds.append(aloop_cmd)
218
219                total_alooped += buf_len_samples
220
221                aenable_cmd = f"between(t,{buf_pos_enable},{buf_len_enable})"
222                aenable_cmds.append(aenable_cmd)
223
224            if int(buf_pos_enable) == 0:
225                self.enable_black_cmd = f"between(t,0,{buf_len_enable_video})"
226
227        # needs an extra trim at the end to get the end of the file
228        if self.fps is None:
229            raise RuntimeError("fps not specified!")
230
231        duration_in_frames = (
232            int(self._get_duration_in_seconds() * self.fps) + total_vlooped
233        )
234        trim_cmd = f"trim=start_frame={last_buf_end}:end_frame={duration_in_frames},setpts=PTS-STARTPTS"
235        trim_cmds.append(trim_cmd)
236
237        self.vloop_cmd = (",").join(vloop_cmds)
238        self.aloop_cmd = (",").join(aloop_cmds)
239        self.trim_cmds = trim_cmds
240        self.venable_cmd = ("+").join(venable_cmds)
241        self.aenable_cmd = ("+").join(aenable_cmds)

314    def trim_video(self):
315        """
316        Remove frames after the frozen, repeated, ones to emulate freezing with skipping
317        """
318        trim_extra_frames = [
319            "ffmpeg",
320            self.overwrite_spec,
321        ]
322
323        trim_extra_frames.extend(
324            [
325                "-i",
326                self._get_tmp_filename("video"),
327            ]
328        )
329
330        vfilters = []
331        filter_interface_list = []
332
333        for ii in range(0, len(self.trim_cmds)):
334            filter_interface_list.append(f"[i{ii}v]")
335
336        remaining_trim_cmds = list(self.trim_cmds)
337        for ii, jj in enumerate(filter_interface_list):
338            filter_string = f"[0:v]{remaining_trim_cmds.pop(0)}{jj}"
339            vfilters.append(filter_string)
340        filters = ";".join(vfilters) + ";"
341        filters += "".join(filter_interface_list)
342        filters += f"concat=n={len(self.trim_cmds)}:v=1[outv]"
343
344        trim_extra_frames.extend(["-filter_complex", filters])
345
346        trim_extra_frames.extend(["-map", "[outv]"])
347
348        trim_extra_frames.extend(
349            [
350                "-c:v",
351                self.vcodec,
352                "-vsync",
353                "cfr",
354                self._get_tmp_filename("skipping"),
355            ]
356        )
357
358        self.run_command(trim_extra_frames)

476    def _get_duration_in_seconds(self):
477        """
478        Convert between the HH:MM:SS.sss format, to total number of seconds.
479        """
480        if self.input_duration is None:

484            datetime.timedelta(
